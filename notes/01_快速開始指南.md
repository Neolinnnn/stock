## 🎯 **第四步：建立你的交易策略**

### 推薦學習路徑

#### **第 1 週：基礎入門**
- 學習 twstock 股價擷取
- 熟悉技術面指標（MA、RSI、MACD）
- 開發第一個簡單策略：移動平均交叉

**目標程式：**
```python
# strategy_ma_crossover.py
from twstock import Stock
import pandas as pd

def ma_crossover_signal(stock_id, days=30):
    """移動平均線交叉策略"""
    stock = Stock(stock_id)
    stock.update()
    
    prices = stock.price[-days:]
    ma5 = sum(prices[-5:]) / 5
    ma20 = sum(prices[-20:]) / 20
    
    current_price = prices[-1]
    
    if ma5 > ma20:
        return "買入信號", current_price
    elif ma5 < ma20:
        return "賣出信號", current_price
    else:
        return "持有", current_price

# 測試
signal, price = ma_crossover_signal('2330')
print(f"台積電信號: {signal} @ 價格: {price}")
```

#### **第 3 週：回測框架**
- 建立歷史資料庫

**基本回測範本：**
```python
import pandas as pd
import numpy as np

def backtest_strategy(stock_data, initial_capital=100000):
    """簡單的回測框架"""
    
    # 初始化
    positions = []
    capital = initial_capital
    pnl_history = []
    
    # 逐日執行策略
    for i in range(len(stock_data)):
        current_price = stock_data.iloc[i]['close']
        signal = generate_signal(stock_data, i)
        
        if signal == "買入" and capital > 0:
            shares = capital / current_price
            positions.append({'entry': current_price, 'shares': shares, 'date': stock_data.iloc[i]['date']})
            capital = 0
        
        elif signal == "賣出" and positions:
            pos = positions.pop()
            pnl = (current_price - pos['entry']) * pos['shares']
            capital = capital + pnl + (pos['entry'] * pos['shares'])
            pnl_history.append(pnl)
    
    # 計算績效指標
    total_return = (capital - initial_capital) / initial_capital
    sharpe_ratio = np.mean(pnl_history) / np.std(pnl_history) if pnl_history else 0
    max_drawdown = calculate_max_drawdown(capital)
    
    return {
        'total_return': total_return,
        'sharpe_ratio': sharpe_ratio,
        'max_drawdown': max_drawdown,
        'trades': len(pnl_history)
    }

# 執行回測
results = backtest_strategy(stock_data)
print(f"報酬率: {results['total_return']:.2%}")
print(f"Sharpe 比率: {results['sharpe_ratio']:.2f}")
```
